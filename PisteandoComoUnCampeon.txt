-- a
esBuenEstado :: Auto -> Bool
esBuenEstado auto = ((< 40) . fst . desgaste) auto  && ((< 60) . snd . desgaste) auto 


-- b 
noDaMas :: Auto -> Bool
noDaMas auto = ((> 80) . fst . desgaste) auto || ((> 80) . snd . desgaste) auto 


-- 3
repararAuto :: Auto -> Auto
repararAuto auto = bajarDesgasteChasis auto

bajarDesgasteChasis :: Auto -> Auto
bajarDesgasteChasis auto = auto{
    desgaste = ( (fst desgaste auto) *0.15 , 0)
}

data Pista = Pista {
    curva :: [],
    rectas :: [],

}

Type Tramo = Auto -> Auto

curva :: Number -> Number -> Tramo
curva angulo longitud auto = auto {
    desgaste = (fst (desgaste auto), (snd desgaste auto) * (3* longitud/angulo) ),
    tiempoDeCarrera = sumarTiempoLongitud longitud auto 2
}

sumarTiempoLongitud :: Number -> Auto -> Number -> Number
sumarTiempoLongitud longitud auto numero = tiempoDeCarrera auto + longitud / (velocidadMaxima auto / numero )

-- b
recto :: Number -> Tramo
recto longitud auto = auto {
    desgaste = desgaste = (fst (desgaste auto) + longitud * 0.01, snd (desgaste auto)),
    tiempoDeCarrera = sumarTiempoLongitud longitud auto 1
}

-- c 
boxes :: Number -> Tramo 
boxes tiempo auto 
 | esBuenEstado auto = sumarTiempo tiempo auto,
 | otherwise = (repararAuto . sumarTiempo (tiempo+10)) auto


sumarTiempo :: Number -> Auto -> Auto
sumarTiempo tiempo auto = auto {
    tiempoDeCarrera = tiempoDeCarrera auto + tiempo
}
