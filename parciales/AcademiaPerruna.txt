data Perro = Perro {
  nombre:: String,
  estamina:: Number,
  juguete:: Juguete,
  criterioJuego :: Perro -> Bool
} 

data Juguete = Juguete {
    color :: String,
    rechina :: Bool
}

recibirJuguete :: Juguete -> Perro -> Perro
recibirJuguete juguete perro = (cambiarEstamina (*1.03)) $ (jugueteNuevo perro) juguete

jugueteNuevo perro nuevo = perro {juguete = nuevo} 
cambiarEstamina x perro = perro {estamina = x (estamina perro)}
